def check_character_legibility():
    # --font-code ensures clear distinction between similar characters.
    
    # Integers and letters: 1, l, I
    integer_one = 1
    letter_l = "l"
    letter_I = "I"
    
    # Zero and the letter O: 0, O
    number_zero = 0
    letter_O = "O"
    
    print(f"Numbers: {integer_one}, {number_zero}")
    print(f"Letters: {letter_l}, {letter_I}, {letter_O}")

check_character_legibility()
